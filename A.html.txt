<script>
    // ----- ИГРОВАЯ ЛОГИКА -----
    let score = 0;
    let gameActive = false;      // игра идёт только после нажатия "Играть"
    let currentDifficulty = 'medium'; // easy, medium, hard
    let timeLeft = 0;
    let timerInterval = null;
    let clickStreak = 0;          // серия кликов без паузы
    let streakTimeout = null;

    // DOM элементы
    const scoreDisplay = document.getElementById('scoreDisplay');
    const clickArea = document.getElementById('clicker');
    const playBtn = document.getElementById('playBtn');
    const exitBtn = document.getElementById('exitBtn');
    const bonusSpan = document.getElementById('bonusMessage');
    const shareBtn = document.getElementById('shareBtn');

    // Настройки сложности (время в секундах, очки за клик)
    const difficultySettings = {
        easy: { time: 45, pointsPerClick: 2, streakBonusThreshold: 10 },
        medium: { time: 30, pointsPerClick: 1, streakBonusThreshold: 15 },
        hard: { time: 20, pointsPerClick: 1, streakBonusThreshold: 20 }
    };

    // Звуки (используем Web Audio, чтобы работало везде)
    function playClickSound() {
        try {
            const audioCtx = new (window.AudioContext || window.webkitAudioContext)();
            const oscillator = audioCtx.createOscillator();
            const gainNode = audioCtx.createGain();
            oscillator.connect(gainNode);
            gainNode.connect(audioCtx.destination);
            oscillator.frequency.value = 880;
            gainNode.gain.value = 0.15;
            oscillator.type = 'sine';
            oscillator.start();
            gainNode.gain.exponentialRampToValueAtTime(0.00001, audioCtx.currentTime + 0.3);
            oscillator.stop(audioCtx.currentTime + 0.2);
            // не закрываем контекст, чтобы звуки были
        } catch(e) { /* без звука */ }
    }

    function playLevelUpSound() {
        try {
            const audioCtx = new (window.AudioContext || window.webkitAudioContext)();
            const osc = audioCtx.createOscillator();
            const gain = audioCtx.createGain();
            osc.connect(gain);
            gain.connect(audioCtx.destination);
            osc.frequency.value = 440;
            gain.gain.value = 0.2;
            osc.type = 'triangle';
            osc.start();
            gain.gain.exponentialRampToValueAtTime(0.00001, audioCtx.currentTime + 0.4);
            osc.stop(audioCtx.currentTime + 0.3);
        } catch(e) {}
    }

    // Обновление счёта на экране
    function updateUI() {
        scoreDisplay.innerText = Math.floor(score);
    }

    // Проверка достижений (бонусы за серию)
    function checkStreakBonus() {
        const settings = difficultySettings[currentDifficulty];
        if (clickStreak >= settings.streakBonusThreshold && clickStreak % settings.streakBonusThreshold === 0) {
            // бонус: дополнительные очки = порог * 0.5
            const bonus = Math.floor(settings.streakBonusThreshold * 0.8);
            score += bonus;
            bonusSpan.innerText = `✨ Бонус за серию ${clickStreak} кликов! +${bonus} ✨`;
            setTimeout(() => {
                if (bonusSpan.innerText.includes('Бонус')) bonusSpan.innerText = '🔥 Бонус: +0';
            }, 1500);
            playLevelUpSound();
            updateUI();
        } else {
            if (!bonusSpan.innerText.includes('Бонус за серию')) {
                bonusSpan.innerText = `🎯 Серия: ${clickStreak} кликов`;
            }
        }
    }

    // Сброс таймера серии
    function resetStreakTimer() {
        if (streakTimeout) clearTimeout(streakTimeout);
        streakTimeout = setTimeout(() => {
            if (clickStreak > 0) {
                clickStreak = 0;
                bonusSpan.innerText = '⚡ Серия прервана! Начни новую ⚡';
                setTimeout(() => {
                    if (bonusSpan.innerText.includes('прервана')) bonusSpan.innerText = '🔥 Бонус: +0';
                }, 1000);
            }
        }, 1800);
    }

    // Обработка клика
    function handleClick() {
        if (!gameActive) {
            // если игра не активна, можно короткий звук предупреждения или ничего
            return;
        }
        const settings = difficultySettings[currentDifficulty];
        score += settings.pointsPerClick;
        clickStreak++;
        resetStreakTimer();
        updateUI();
        playClickSound();
        checkStreakBonus();

        // Анимация нажатия (добавляем ripple эффект)
        const clickerDiv = document.getElementById('clicker');
        clickerDiv.style.transform = 'scale(0.96)';
        setTimeout(() => { clickerDiv.style.transform = ''; }, 100);
    }

    // Таймер обратного отсчёта
    function startTimer() {
        if (timerInterval) clearInterval(timerInterval);
        timerInterval = setInterval(() => {
            if (!gameActive) return;
            if (timeLeft <= 0) {
                // игра окончена
                clearInterval(timerInterval);
                timerInterval = null;
                gameActive = false;
                alert(`Время вышло! Ваш результат: ${Math.floor(score)} очков. Нажмите "Играть" для новой игры.`);
                playBtn.innerText = '▶ Играть';
            } else {
                timeLeft--;
                // отображаем время в счёте? можно добавить отдельный индикатор, но для минимализма покажем в заголовке страницы
                document.title = `Кликер | ${timeLeft} сек | ${Math.floor(score)} очков`;
                if (timeLeft === 0) {
                    document.title = 'Кликер';
                }
            }
        }, 1000);
    }

    // Запуск / сброс игры
    function startNewGame() {
        if (timerInterval) clearInterval(timerInterval);
        gameActive = true;
        score = 0;
        clickStreak = 0;
        if (streakTimeout) clearTimeout(streakTimeout);
        bonusSpan.innerText = '🔥 Бонус: +0';
        const settings = difficultySettings[currentDifficulty];
        timeLeft = settings.time;
        updateUI();
        startTimer();
        playBtn.innerText = '🔁 Играть (новая)';
        document.title = `Кликер | ${timeLeft} сек | 0 очков`;
        // маленький звук старта
        playLevelUpSound();
    }

    function exitGame() {
        if (timerInterval) clearInterval(timerInterval);
        gameActive = false;
        score = 0;
        clickStreak = 0;
        updateUI();
        playBtn.innerText = '▶ Играть';
        document.title = 'Кликер';
        bonusSpan.innerText = '🔥 Бонус: +0';
        if (streakTimeout) clearTimeout(streakTimeout);
        alert('Выход из игры. Счёт сброшен.');
    }

    // Смена сложности (только когда игра не активна)
    function setDifficulty(level) {
        if (gameActive) {
            alert('Нельзя сменить сложность во время игры. Нажмите "Выход" или закончите раунд.');
            return;
        }
        currentDifficulty = level;
        const settings = difficultySettings[level];
        bonusSpan.innerText = `🎚 Сложность: ${level === 'easy' ? 'Лёгкая (+2 очка за клик)' : (level === 'medium' ? 'Средняя' : 'Сложная')}`;
        setTimeout(() => {
            if (bonusSpan.innerText.includes('Сложность')) bonusSpan.innerText = '🔥 Бонус: +0';
        }, 2000);
    }

    // Поделиться результатом (Web Share API)
    function shareResult() {
        if (!gameActive && score === 0) {
            alert('Сыграйте сначала, чтобы поделиться результатом.');
            return;
        }
        const text = `Я набрал ${Math.floor(score)} очков в игре-кликере на сложности ${currentDifficulty}! Попробуй меня побить!`;
        if (navigator.share) {
            navigator.share({
                title: 'Мой рекорд в кликере',
                text: text,
                url: window.location.href
            }).catch(() => alert('Не удалось поделиться'));
        } else {
            // fallback: копирование в буфер
            navigator.clipboard.writeText(text);
            alert('Результат скопирован в буфер обмена: ' + text);
        }
    }

    // ----- Привязка событий (с поддержкой touch) -----
    clickArea.addEventListener('click', handleClick);
    clickArea.addEventListener('touchstart', (e) => {
        e.preventDefault();
        handleClick();
    });
    playBtn.addEventListener('click', startNewGame);
    exitBtn.addEventListener('click', exitGame);
    shareBtn.addEventListener('click', shareResult);

    document.querySelectorAll('[data-diff]').forEach(btn => {
        btn.addEventListener('click', (e) => {
            const level = e.target.getAttribute('data-diff');
            setDifficulty(level);
        });
    });

    // Инициализация (игра не активна, но интерфейс готов)
    setDifficulty('medium');
    updateUI();
    console.log('Игра загружена. Нажмите "Играть"');
</script>
